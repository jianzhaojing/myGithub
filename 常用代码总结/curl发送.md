        curl_setopt($ch,CURLOPT_RETURNTRANSFER,1);

        curl_setopt($ch,CURLOPT_HTTPHEADER,$header);
        //发送post请求
        curl_setopt($ch,CURLOPT_POST,1);
        curl_setopt($ch,CURLOPT_POSTFIELDS,[
            'title'=>'curl2',
            'author'=>'jerry',
            'desc'=>'hhhhhhhhhh'
        ]);
        //执行命令
        $data = curl_exec($ch);
        //关闭curl
        curl_close($ch);
